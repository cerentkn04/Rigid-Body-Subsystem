constexpr int WINDOW_WIDTH = 800;

enum class CellType : uint8_t {
    Empty = 0, Sand, Water, Wall
};

struct Cell {
    CellType type = CellType::Empty;
    bool updated = false;
    uint8_t colorVariation = 0;
};

// -----------------------------------------------------------------------------
// STANDALONE EXTRACTOR (Used for Testing without changing HPPs)
// -----------------------------------------------------------------------------
struct TestExtractor {
    std::vector<rigid::RegionIndex> label_grid;
    std::vector<rigid::RegionBuildRecord> build_records;
    std::vector<int> work_queue_x;
    std::vector<int> work_queue_y;

    void extract(int width, int height, const std::vector<Cell>& grid) {
        int total = width * height;
        if (label_grid.size() != (size_t)total) label_grid.resize(total);
        
        std::fill(label_grid.begin(), label_grid.end(), rigid::InvalidRegionIndex);
        build_records.clear();

        for (int y = 0; y < height; ++y) {
            for (int x = 0; x < width; ++x) {
                if (grid[y * width + x].type == CellType::Wall && 
                    label_grid[y * width + x] == rigid::InvalidRegionIndex) {
                    
                    floodFill(x, y, width, height, grid);
                }
            }
        }
    }

private:
    void floodFill(int sx, int sy, int w, int h, const std::vector<Cell>& grid) {
        rigid::RegionIndex current_idx = (rigid::RegionIndex)build_records.size();
        rigid::RegionBuildRecord record{sx, sy, sx, sy, 0};
        
        work_queue_x.clear(); work_queue_y.clear();
        
        auto claim = [&](int x, int y) {
            label_grid[y * w + x] = current_idx;
            record.pixel_count++;
            record.min_x = std::min(record.min_x, x);
            record.max_x = std::max(record.max_x, x);
            record.min_y = std::min(record.min_y, y);
            record.max_y = std::max(record.max_y, y);
            work_queue_x.push_back(x);
            work_queue_y.push_back(y);
        };

        claim(sx, sy);

        while (!work_queue_x.empty()) {
            int cx = work_queue_x.back(); work_queue_x.pop_back();
            int cy = work_queue_y.back(); work_queue_y.pop_back();

            for (int dy = -1; dy <= 1; ++dy) {
                for (int dx = -1; dx <= 1; ++dx) {
                    if (dx == 0 && dy == 0) continue;
                    int nx = cx + dx, ny = cy + dy;
                    if (nx >= 0 && nx < w && ny >= 0 && ny < h) {
                        if (grid[ny * w + nx].type == CellType::Wall && 
                            label_grid[ny * w + nx] == rigid::InvalidRegionIndex) {
                            claim(nx, ny);
                        }
                    }
                }
            }
        }
        build_records.push_back(record);
    }
};

class SandSimulation {
public:
    std::vector<Cell> grid;
    std::mt19937 rng;
    SandSimulation() : grid(GRID_WIDTH * GRID_HEIGHT), rng(std::random_device{}()) {}
    Cell& at(int x, int y) { return grid[y * GRID_WIDTH + x]; }
    const Cell& at(int x, int y) const { return grid[y * GRID_WIDTH + x]; }
    bool inBounds(int x, int y) const { return x >= 0 && x < GRID_WIDTH && y >= 0 && y < GRID_HEIGHT; }
    bool isEmpty(int x, int y) const { return inBounds(x, y) && at(x, y).type == CellType::Empty; }
    bool canDisplace(int x, int y, CellType displacing) const {
        if (!inBounds(x, y)) return false;
        CellType target = at(x, y).type;
        return (target == CellType::Empty) || (displacing == CellType::Sand && target == CellType::Water);
    }
    void swap(int x1, int y1, int x2, int y2) {
        std::swap(at(x1, y1), at(x2, y2));
        at(x1, y1).updated = true; at(x2, y2).updated = true;
    }
    void updateSand(int x, int y) {
        if (at(x, y).updated) return;
        if (canDisplace(x, y + 1, CellType::Sand)) { swap(x, y, x, y + 1); return; }
        bool canL = canDisplace(x - 1, y + 1, CellType::Sand), canR = canDisplace(x + 1, y + 1, CellType::Sand);
        if (canL && canR) swap(x, y, x + ((rng() % 2) ? -1 : 1), y + 1);
        else if (canL) swap(x, y, x - 1, y + 1);
        else if (canR) swap(x, y, x + 1, y + 1);
    }
    void updateWater(int x, int y) {
        if (at(x, y).updated) return;
        if (isEmpty(x, y + 1)) { swap(x, y, x, y + 1); return; }
        bool cDL = isEmpty(x - 1, y + 1), cDR = isEmpty(x + 1, y + 1);
        if (cDL && cDR) { swap(x, y, x + ((rng() % 2) ? -1 : 1), y + 1); return; }
        else if (cDL) { swap(x, y, x - 1, y + 1); return; }
        else if (cDR) { swap(x, y, x + 1, y + 1); return; }
        bool cL = isEmpty(x - 1, y), cR = isEmpty(x + 1, y);
        if (cL && cR) swap(x, y, x + ((rng() % 2) ? -1 : 1), y);
        else if (cL) swap(x, y, x - 1, y);
        else if (cR) swap(x, y, x + 1, y);
    }
    void update() {
        for (auto& cell : grid) cell.updated = false;
        for (int y = GRID_HEIGHT - 1; y >= 0; --y) {
            bool LtoR = (rng() % 2) == 0;
            for (int i = 0; i < GRID_WIDTH; ++i) {
                int x = LtoR ? i : (GRID_WIDTH - 1 - i);
                if (at(x, y).type == CellType::Sand) updateSand(x, y);
                else if (at(x, y).type == CellType::Water) updateWater(x, y);
            }
        }
    }
    void placeCell(int x, int y, CellType type, int brushSize = 1) {
        std::uniform_int_distribution<uint8_t> dist(0, 30);
        for (int dy = -brushSize + 1; dy < brushSize; ++dy) {
            for (int dx = -brushSize + 1; dx < brushSize; ++dx) {
                int nx = x + dx, ny = y + dy;
                if (inBounds(nx, ny)) {
                    if (type == CellType::Empty || at(nx, ny).type == CellType::Empty) {
                        at(nx, ny).type = type; at(nx, ny).colorVariation = dist(rng);
                    }
                }
            }
        }
    }
    void clear() { for (auto& cell : grid) { cell.type = CellType::Empty; cell.colorVariation = 0; } }
};

SDL_Color getCellColor(const Cell& cell) {
    int v = cell.colorVariation;
    switch (cell.type) {
        case CellType::Sand: return { (uint8_t)(220 - v), (uint8_t)(180 - v), (uint8_t)(80 - v), 255 };
        case CellType::Water: return { (uint8_t)(30 + v), (uint8_t)(100 + v), (uint8_t)(200 + v), 200 };
        case CellType::Wall: return { (uint8_t)(100 - v), (uint8_t)(100 - v), (uint8_t)(100 - v), 255 };
        default: return { 20, 20, 30, 255 };
    }
}
